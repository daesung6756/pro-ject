if 조건문

if 조건문이란? 특정 조건을 설정하여, 해당 조건이 충족되면 작성해둔 이벤트가 발생하는 구문이다.
if(조건){}, else if(조건){}, else {} 이렇게 3가지로 분류된다.

ex) if문만 사용시
var index = 1; // index 변수의 값은 1이다.
if ( index == 1 ) { // 만약에 index 값이 1과 같다면
alert("안녕하세요"); // 경고창에 "안녕하세요"라는 문자를 나타내라
}
* 위의 코드를 실행해 보면 경고창에 "안녕하세요" 라는 메세지가 나올 것이다.


ex) if, else 사용시
var index = 2; // index 변수의 값은 2이다.
if ( index == 1 ) { // 만약에 index 값이 1과 같다면
alert("안녕하세요"); // 경고창에 "안녕하세요"라는 문자를 나타내라
} else { // 만약 1이 아니라면 아래 이벤트를 실행해라
alert("어서오세요");// 경고창에 "어서오세요"라는 문자를 나타내라
}
* 위의 코드를 실행해 보면 경고창에 "어서오세요" 라는 메세지가 나올 것이다.


ex) if, else if, else 모두 사용시
var index = 4; // index 변수의 값은 4이다.
if ( index == 1 ) { // 만약에 index 값이 1과 같다면
alert("안녕하세요"); // 경고창에 "안녕하세요"라는 문자를 나타내라
} else if (index == 3 ){ // 만약 index가 3과 같다면
alert("어서오세요");// 경고창에 "어서오세요" 라는 문자를 나타내라
} else { // index가 1과 3이 아니라면 아래 이벤트를 실행해라
alert("반갑습니다."); // 경고창에 "반갑습니다" 라는 문자를 나타내라
}
* 위의 코드를 실행해 보면 경고창에 "반갑습니다" 라는 메세지가 나올 것이다.


else 는 if, else if를 제외한 나머지 조건에 해당 한다.
